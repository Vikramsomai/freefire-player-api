javascript

          
            fetch('https://vikramsomai.github.io/freefire-player-api/player.json')
           .then(function (response) {
                return response.json();
             })
             .then(function (data) {
               console.log(data);
             })
             .catch(function (err) {
                 console.log("Something went wrong!", err);
             });
